dynamic toJson (Weekday input)

Converts a Weekday to JSON.

Implementation 

static dynamic toJson(Weekday input) {
  switch (input) {
    case Weekday.Monday:
      return 0;
      break;

    case Weekday.Tuesday:
      return 1;
      break;

    case Weekday.Wednesday:
      return 2;
      break;

    case Weekday.Thursday:
      return 3;
      break;

    case Weekday.Friday:
      return 4;
      break;
  }
}
